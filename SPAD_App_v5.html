function doPost(e) {
  try {
    var sheet = SpreadsheetApp
      .getActiveSpreadsheet().getActiveSheet();
    if (sheet.getLastRow() === 0) {
      sheet.appendRow([
        "Timestamp","Sales Rep","Farmer",
        "Date","Time","GPS Lat","GPS Lon",
        "Rainfall 14d (mm)","Device ID","Measurement ID",
        "Crop","Variety","FAO No","Growth Stage","Leaf Position",
        "atLEAF Avg","N Rate (kg/ha)","N Appl Date",
        "Soil Type","Prev Crop","Visual Obs","AI Interpretation"
      ]);
    }
    var d = JSON.parse(e.postData.contents);
    sheet.appendRow([
      new Date().toISOString(),
      d.sales_rep, d.farmer, d.date, d.time, d.lat, d.lon,
      d.rainfall, d.device_id, d.meas_id,
      d.crop, d.variety, d.fao, d.stage, d.leaf_pos,
      d.atleaf_avg, d.n_rate, d.n_date,
      d.soil, d.prev_crop, d.visual, d.ai_interpretation
    ]);
    return ContentService
      .createTextOutput(JSON.stringify({status:"ok"}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({status:"error",msg:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({status:"alive"}))
    .setMimeType(ContentService.MimeType.JSON);
}
